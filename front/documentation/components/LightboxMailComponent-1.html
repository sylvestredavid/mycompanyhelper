import {Component, OnInit} from '@angular/core';
import {FormControl, FormGroup, Validators} from '@angular/forms';
import {UsersService} from '../../users.service';
import {MatDialogRef} from '@angular/material';

@Component({
    selector: 'app-lightbox-mail',
    templateUrl: './lightbox-mail.component.html',
    styleUrls: ['./lightbox-mail.component.scss']
})
export class LightboxMailComponent implements OnInit {

    msg: string;
    mailForm = new FormGroup({
        mail: new FormControl('', Validators.required)
    });

    constructor(private userService: UsersService, public dialogRef: MatDialogRef<LightboxMailComponent>) {
    }

    ngOnInit() {
    }

    /**
     * envoi du mail de demande de changement de mdp, on hash l'email pour pas qu'il soit en clair dan l'url
     */
    onSubmit() {
        const hash = btoa(this.mailForm.value.mail)
        this.userService.mailPassword(this.mailForm.value.mail, hash).subscribe(
            () => {
                this.msg = 'un mail viens de vous etre envoyé.';
            },
            () => this.msg = 'aucun utilisateur trouvée avec cette adresse mail.'
        );
    }

    /**
     * fermeture de la lightbox
     */
    retour() {
        this.dialogRef.close();
    }

}

<mat-dialog-content>
    <form [formGroup]="mailForm">
          <mat-form-field>
            <input matInput placeholder="email" formControlName="mail">
          </mat-form-field>
        <ng-container *ngIf="msg"><p>{{msg}}</p></ng-container>
    </form>
</mat-dialog-content>
<mat-dialog-actions>
  <button mat-raised-button (click)="onSubmit()">ok</button>
    <button mat-raised-button (click)="retour()">Retour</button>
</mat-dialog-actions>
