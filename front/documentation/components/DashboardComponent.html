import {Component, OnInit} from '@angular/core';
import {ProduitService} from '../produits/produit.service';
import {ProduitModel} from '../../models/produit.model';
import {ClientsService} from '../clients/clients.service';
import {ClientModel} from '../../models/client.model';
import {GenreModel} from '../../models/genre.model';
import {GenreService} from '../genres/genre.service';

@Component({
    selector: 'app-dashboard',
    templateUrl: './dashboard.component.html',
    styleUrls: ['./dashboard.component.scss']
})
export class DashboardComponent implements OnInit {

    listeProduits: ProduitModel[];
    listeClients: ClientModel[];
    listeGenres: GenreModel[];

    constructor(private produitService: ProduitService, private clientService: ClientsService, private genreService: GenreService) {
    }

    ngOnInit() {

        this.produitService.publishProduits();
        this.produitService.listeProduits$.subscribe(
            produits => {
                this.listeProduits = [];
                if (produits) {
                    produits.forEach(
                        produit => {
                            if (produit.enVente === true) {
                                this.listeProduits.push(produit);
                            }
                        }
                    );
                }
            }
        );
        this.clientService.publishClients();
        this.clientService.listeClients$.subscribe(
            clients => this.listeClients = clients
        );
        this.genreService.listeGenre$.subscribe(
            genres => this.listeGenres = genres
        );
    }

}
