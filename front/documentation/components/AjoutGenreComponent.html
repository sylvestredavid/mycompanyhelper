import {Component, OnDestroy, OnInit} from '@angular/core';
import {FormControl, FormGroup, Validators} from '@angular/forms';
import {GenreService} from '../genre.service';
import {MatDialogRef} from '@angular/material';
import {GenreModel} from '../../../models/genre.model';
import {UsersService} from '../../users/users.service';

@Component({
    selector: 'app-ajout-genre',
    templateUrl: './ajout-genre.component.html',
    styleUrls: ['./ajout-genre.component.scss']
})
export class MenuFournisseurComponent implements OnInit, OnDestroy {

    genreForm: FormGroup;

    constructor(public dialogRef: MatDialogRef<MenuFournisseurComponent>, private genreService: GenreService,
                private userService: UsersService) {
    }

    ngOnInit() {
        this.initForm();
    }

    /**
     * initialisation du formulaire
     */
    initForm(){
        this.genreForm = new FormGroup({
            designation: new FormControl('', Validators.required)
        });
    }

    /**
     * envoi du formulaire
     */
    onSave() {
        const genre: GenreModel = {
            idGenre: null,
            designation: this.genreForm.value.designation,
            produits: null,
            idUser: this.userService.idUser
        };
        this.genreService.saveGenre(genre);
        this.dialogRef.close();
    }

    onQuit() {
        this.dialogRef.close();
    }

    ngOnDestroy(): void {
    }

}

<mat-dialog-content>
  <h1>Ajouter categorie</h1>
    <form [formGroup]="genreForm">
          <mat-form-field>
            <input matInput placeholder="nom" formControlName="designation">
          </mat-form-field>
    </form>
</mat-dialog-content>
<mat-dialog-actions>
  <button mat-raised-button [disabled]="genreForm.invalid" (click)="onSave()">Envoyer</button>
  <button mat-raised-button (click)="onQuit()">Quitter</button>
</mat-dialog-actions>
