import {Component, OnInit} from '@angular/core';
import {ActivatedRoute} from '@angular/router';
import {FormBuilder, FormGroup, Validators} from '@angular/forms';
import {EmailModel} from '../../../models/email.model';
import {AnnuaireService} from '../annuaire.service';
import {Store} from '@ngrx/store';
import {UserState} from '../../../shared/stores/user.reducer';
import {MatSnackBar} from '@angular/material';

@Component({
    selector: 'app-contact-fournisseur',
    templateUrl: './contact-fournisseur.component.html',
    styleUrls: ['./contact-fournisseur.component.scss']
})
export class ContactFournisseurHorsConnexionComponent implements OnInit {

    titre: string;
    emailTo: string;
    emailFrom = '';
    mailForm: FormGroup;

    constructor(private route: ActivatedRoute, private fb: FormBuilder, private annuaireService: AnnuaireService,
                private store: Store<UserState>, private  snackBar: MatSnackBar) {
        this.store.select('user').subscribe(
            user => {
                if (user) {
                    this.emailFrom = user[0].username;
                }
            }
        );
    }

    ngOnInit() {
        this.initForm();
    }

    /**
     * envoi du formulaire
     */
    onSubmit() {
        const mail: EmailModel = {
            to: this.emailTo,
            from: this.mailForm.value.email,
            titre: `Votre annonce "${this.titre}" sur powersell.`,
            corps: this.mailForm.value.corps
        };

        this.annuaireService.sendMail(mail).subscribe(
            () => this.snackBar.open('L\'email a bien été envoyé.' , 'ok', {duration: 1500})
            );
    }

    /**
     * initialisation du formulaire
     */
    private initForm() {
        this.emailTo = this.route.snapshot.params.email;
        this.route.queryParams.subscribe(
            params => this.titre = params.annonce
        );

        this.mailForm = this.fb.group({
            email: [this.emailFrom, Validators.compose([Validators.required, Validators.email])],
            corps: ['', Validators.required]
        });
    }
}
