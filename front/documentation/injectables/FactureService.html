import {Injectable} from '@angular/core';
import {HttpClient, HttpHeaders} from '@angular/common/http';
import {FactureModel} from '../../models/facture.model';
import {Observable, Subscription} from 'rxjs';
import {CAModel} from '../../models/ca.model';
import {UsersService} from '../users/users.service';
import {ProduitModel} from '../../models/produit.model';

@Injectable({
    providedIn: 'root'
})
export class FactureService {

    constructor(private http: HttpClient, private userService: UsersService) {
    }

    getOptions() {
        const headerDict = {
            'Content-Type': 'application/json',
            'Authorization': sessionStorage.getItem('token')
        };

        const requestOptions = {
            headers: new HttpHeaders(headerDict),
        };

        return requestOptions;
    }

    saveFacture(facture: FactureModel): Observable<FactureModel> {
        return this.http.post<FactureModel>('http://powersell-back.eu-west-3.elasticbeanstalk.com/api/factures/save', facture, this.getOptions());
    }

    saveProduitsFacture(quantite: number, idFacture: number, idProduit: number) {
        this.http.post('http://powersell-back.eu-west-3.elasticbeanstalk.com/api/factures/saveProduits?quantite=' + quantite + '&idProduit=' + idProduit + '&idFacture='
            + idFacture, '', this.getOptions()).subscribe();
    }

    sendMail(idFacture: number) {
        this.http.post('http://powersell-back.eu-west-3.elasticbeanstalk.com/api/factures/sendMail?idFacture=' + idFacture, '', this.getOptions()).subscribe();
    }

    sendMailStockBas(email: string, produit: string) {
        this.http.post('http://powersell-back.eu-west-3.elasticbeanstalk.com/api/produits/mailStockBas?email=' + email + '&produit=' + produit, '', this.getOptions()).subscribe();
    }

    getCA(): Observable<CAModel[]> {
        return this.http.get<CAModel[]>('http://powersell-back.eu-west-3.elasticbeanstalk.com/api/ca?idUser=' + this.userService.idUser, this.getOptions());
    }
}
