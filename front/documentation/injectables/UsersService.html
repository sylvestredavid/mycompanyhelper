import {Injectable} from '@angular/core';
import {HttpClient, HttpHeaders} from '@angular/common/http';
import {Observable, Subscription} from 'rxjs';
import {UserModel} from '../../models/user.modele';
import {LoginModel} from '../../models/Login.model';
import {SignUpModel} from '../../models/signUp.model';
import {text} from '@angular/core/src/render3';

@Injectable({
    providedIn: 'root'
})
export class UsersService {

    idUser: number;

    constructor(private http: HttpClient) {
    }


    signin(login: LoginModel): Observable<UserModel> {
        return this.http.post<UserModel>('http://powersell-back.eu-west-3.elasticbeanstalk.com/api/users/signin', login);
    }

    signup(request: SignUpModel): Observable<UserModel> {
        return this.http.post<UserModel>('http://powersell-back.eu-west-3.elasticbeanstalk.com/api/users/signup', request);
    }

    findGestionnaires(managementId: number): Observable<UserModel[]>{
        return this.http.get<UserModel[]>(`http://powersell-back.eu-west-3.elasticbeanstalk.com/api/users/${managementId}/gestionnaires`);
    }

    delete(id: number){
        this.http.delete<UserModel[]>(`http://powersell-back.eu-west-3.elasticbeanstalk.com/api/users/gestionnaires/delete?id=${id}`).subscribe();
    }

    changePassword(password: string, email: string){
        const body: LoginModel = {
            username: email,
            password: password
        };
        const url = `http://powersell-back.eu-west-3.elasticbeanstalk.com/api/users/changePassword`;
        this.http.post(url, body).subscribe();
    }

    mailPassword(mail: string, lien: string) {
        return this.http.post(`http://powersell-back.eu-west-3.elasticbeanstalk.com/api/users/mailPassword?mail=${mail}&lien=${lien}`, '');
    }
}
